# tests/snippets_statusbar.txt — expect_statusbar smoke test
#
# SB_GETTEXTA(partN) on a window's msctls_statusbar32 child.
# Signature: ( "title" partN "expected" -- )
# partN is 0-based (matches Win32 SB_GETTEXTA wParam).
#
# Setup before running:
#   - Notepad++ open with at least one file/tab
#   - Status bar visible (default; View menu if hidden)
#
# What Notepad++ shows in its status bar parts (typical):
#   part 0 — current document type (e.g. "Normal text file")
#   part 1 — total / cursor position info
#   part 2 — line/col
#   part 3 — encoding
#   part 4 — EOL mode
#
# Launch:
#   color_iw.exe tests/snippets_statusbar.txt
#
# Run:
#   Ctrl+Shift+B  — part 0 should contain "text file"  (expect PASS)
#   Ctrl+Shift+N  — bad-part fishing for missing text  (expect FAIL=text)
#   Ctrl+Shift+M  — wrong window title                 (expect FAIL=win)
#
# Expected log per call:
#   [ESB] title="..." part=00000000 parent=NNNN sbar=NNNN got="..." want="..." PASS
#   or
#   [TEST ...] FAIL: expect_statusbar: text not found in part, got="..."
#   [ESB] ... FAIL=text

::: tESB_pass ! Ctrl+Shift+B
"statusbar_part0_normal" tname
"new 27 - Notepad++" 0 "text file" expect_statusbar

::: tESB_fail_text ! Ctrl+Shift+N
"statusbar_part0_missing" tname
"new 27 - Notepad++" 0 "zzzz_definitely_not_in_status_bar_zzzz" expect_statusbar

::: tESB_fail_win ! Ctrl+Shift+M
"statusbar_bad_window" tname
"NoSuchWindow_xyz123" 0 "anything" expect_statusbar

# Alternative target — classic Notepad. More reliable than Notepad++
# (title is stable, doesn't change with tab switches). Setup:
#   1. Open classic notepad.exe
#   2. View > Status Bar (enable if not already shown)
#   3. Type any character so position info appears (e.g. "Ln 1, Col 2")
#   4. Title becomes "*Untitled - Notepad" (note the asterisk)
::: tESB_notepad ! Ctrl+Shift+H
"statusbar_notepad_ln" tname
"*Untitled - Notepad" 0 "Ln" expect_statusbar
